# 2. User Personas

## 2.1 Player Personas

### Casual Player (Alex)
- **Background:** Curious about D&D from Stranger Things/Critical Role
- **Goals:** Experience D&D without reading 300+ pages of rules
- **Pain Points:** Intimidated by complexity, doesn't have a local group
- **Needs:** Guided experience, pre-built characters, matchmaking

### Experienced Player (Jordan)
- **Background:** Plays tabletop D&D weekly, wants digital option
- **Goals:** Play with friends remotely, accurate RAW experience
- **Pain Points:** Most digital tools don't follow proper rules
- **Needs:** Full character customization, accurate mechanics, multiplayer

### Content Creator (Casey)
- **Background:** Streams D&D content, creates homebrew
- **Goals:** Create engaging content, build audience
- **Pain Points:** VTTs are visually boring for streams
- **Needs:** Cinematic visuals, campaign sharing, stream integration

## 2.2 Dungeon Master Personas

### New DM (Riley)
- **Background:** Wants to try DMing but intimidated
- **Goals:** Run first campaign without embarrassment
- **Pain Points:** Prep time, rule lookups, encounter balance
- **Needs:** Pre-built campaigns, AI assistance, rule automation

### Veteran DM (Sam)
- **Background:** 10+ years running campaigns
- **Goals:** Create immersive worlds with visual storytelling
- **Pain Points:** Limited by tabletop constraints
- **Needs:** Deep customization, cutscene creation, full control

## 5.2 Touch Gestures

```typescript
// hooks/useTouchGestures.ts

interface GestureConfig {
  onTap?: (position: Point) => void;
  onDoubleTap?: (position: Point) => void;
  onLongPress?: (position: Point) => void;
  onPan?: (delta: Point) => void;
  onPinch?: (scale: number, center: Point) => void;
  onSwipe?: (direction: 'up' | 'down' | 'left' | 'right') => void;
}

export function useTouchGestures(ref: RefObject<HTMLElement>, config: GestureConfig) {
  useEffect(() => {
    const element = ref.current;
    if (!element) return;

    const hammer = new Hammer(element);
    
    // Configure recognizers
    hammer.get('pinch').set({ enable: true });
    hammer.get('pan').set({ direction: Hammer.DIRECTION_ALL });
    
    // Event handlers
    if (config.onTap) {
      hammer.on('tap', (e) => config.onTap!({ x: e.center.x, y: e.center.y }));
    }
    
    if (config.onDoubleTap) {
      hammer.on('doubletap', (e) => config.onDoubleTap!({ x: e.center.x, y: e.center.y }));
    }
    
    if (config.onLongPress) {
      hammer.on('press', (e) => config.onLongPress!({ x: e.center.x, y: e.center.y }));
    }
    
    if (config.onPan) {
      hammer.on('pan', (e) => config.onPan!({ x: e.deltaX, y: e.deltaY }));
    }
    
    if (config.onPinch) {
      hammer.on('pinch', (e) => config.onPinch!(e.scale, { x: e.center.x, y: e.center.y }));
    }
    
    if (config.onSwipe) {
      hammer.on('swipe', (e) => {
        const dir = e.direction === Hammer.DIRECTION_UP ? 'up' :
                    e.direction === Hammer.DIRECTION_DOWN ? 'down' :
                    e.direction === Hammer.DIRECTION_LEFT ? 'left' : 'right';
        config.onSwipe!(dir);
      });
    }

    return () => hammer.destroy();
  }, [ref, config]);
}
```

// services/tutorial/TutorialManager.ts

interface TutorialStep {
  id: string;
  type: 'highlight' | 'modal' | 'tooltip' | 'guided_action';
  target?: string;  // CSS selector
  title: string;
  content: string;
  action?: {
    type: 'click' | 'drag' | 'type' | 'select';
    validation: string;  // What to check
  };
  nextTrigger: 'click' | 'action_complete' | 'timer';
  skippable: boolean;
}

interface Tutorial {
  id: string;
  name: string;
  context: 'game' | 'character_builder' | 'campaign_builder' | 'general';
  prerequisite?: string;  // Previous tutorial ID
  steps: TutorialStep[];
  reward?: {
    type: 'badge' | 'item' | 'xp';
    value: string;
  };
}

const TUTORIALS: Tutorial[] = [
  // Core gameplay
  {
    id: 'movement_basics',
    name: 'Movement Basics',
    context: 'game',
    steps: [
      {
        id: 'select_character',
        type: 'highlight',
        target: '.token-player',
        title: 'Your Character',
        content: 'This is your character token. Tap to select it.',
        nextTrigger: 'click',
        skippable: false,
      },
      {
        id: 'show_movement',
        type: 'highlight',
        target: '.tile-highlighted',
        title: 'Movement Range',
        content: 'The highlighted tiles show where you can move. Your speed is 30 feet (6 tiles).',
        nextTrigger: 'click',
        skippable: true,
      },
      {
        id: 'move_character',
        type: 'guided_action',
        target: '.tile-target',
        title: 'Move Here',
        content: 'Tap the marked tile to move your character.',
        action: { type: 'click', validation: 'character_moved' },
        nextTrigger: 'action_complete',
        skippable: false,
      },
    ],
    reward: { type: 'badge', value: 'first_steps' },
  },
  
  // Combat
  {
    id: 'combat_basics',
    name: 'Combat 101',
    context: 'game',
    prerequisite: 'movement_basics',
    steps: [
      {
        id: 'explain_initiative',
        type: 'modal',
        title: 'Initiative',
        content: 'Combat has begun! Everyone rolls initiative to determine turn order. The highest roll goes first.',
        nextTrigger: 'click',
        skippable: true,
      },
      {
        id: 'select_attack',
        type: 'highlight',
        target: '#action-attack',
        title: 'Attack Action',
        content: 'Tap the sword icon to make an attack.',
        nextTrigger: 'click',
        skippable: false,
      },
      // ... more steps
    ],
  },
];

export class TutorialManager {
  private currentTutorial: Tutorial | null = null;
  private currentStepIndex: number = 0;
  private completedTutorials: Set<string> = new Set();

  async startTutorial(tutorialId: string): Promise<void> {
    const tutorial = TUTORIALS.find(t => t.id === tutorialId);
    if (!tutorial) throw new Error(`Tutorial not found: ${tutorialId}`);
    
    // Check prerequisite
    if (tutorial.prerequisite && !this.completedTutorials.has(tutorial.prerequisite)) {
      throw new Error(`Must complete ${tutorial.prerequisite} first`);
    }
    
    this.currentTutorial = tutorial;
    this.currentStepIndex = 0;
    
    await this.showStep(tutorial.steps[0]);
  }

  private async showStep(step: TutorialStep): Promise<void> {
    switch (step.type) {
      case 'highlight':
        this.highlightElement(step.target!, step.title, step.content);
        break;
      case 'modal':
        await this.showModal(step.title, step.content);
        break;
      case 'tooltip':
        this.showTooltip(step.target!, step.content);
        break;
      case 'guided_action':
        await this.guideAction(step);
        break;
    }
  }

  async completeStep(): Promise<void> {
    if (!this.currentTutorial) return;
    
    this.currentStepIndex++;
    
    if (this.currentStepIndex >= this.currentTutorial.steps.length) {
      await this.completeTutorial();
    } else {
      await this.showStep(this.currentTutorial.steps[this.currentStepIndex]);
    }
  }

  private async completeTutorial(): Promise<void> {
    if (!this.currentTutorial) return;
    
    this.completedTutorials.add(this.currentTutorial.id);
    
    // Award reward
    if (this.currentTutorial.reward) {
      await this.awardReward(this.currentTutorial.reward);
    }
    
    // Save progress
    await this.saveProgress();
    
    this.currentTutorial = null;
  }
}
